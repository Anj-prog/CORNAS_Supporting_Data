allmale <- read.table("allgenes.male.count_table.txt",sep=" ",header=TRUE)
allfemale <- read.table("allgenes.female.count_table.txt",sep=" ",header=TRUE)
for (i in 1:100) {
	fem <- allfemale[,z[i,1]+1] #female
	mal <- allmale[,z[i,2]+1] #male
	table2print <- data.frame(allmale[,1],mal,fem)
	write.table(table2print, file=paste("comparisons_allgenes/compare_all",z[i,1],z[i,2],"output.tab",sep="_"),sep="\t",row.names=FALSE,col.names=FALSE)
}
